# Progress UI

Display throughput and memory usage to build user trust.

## Minimal example

<SandpackExample
  template="vanilla-ts"
  activeFile="/index.ts"
  preview={false}
  files={{
    "/index.ts": `
import { ConvertBuddy } from "convert-buddy-js";

const buddy = new ConvertBuddy({
  onProgress: (stats) => setStats(stats)
});
`,
  }}
/>


## Advanced example

<SandpackExample
  template="vanilla-ts"
  activeFile="/index.ts"
  preview={false}
  files={{
    "/index.ts": `
import { ConvertBuddy } from "convert-buddy-js";

const buddy = new ConvertBuddy({
  onProgress: (stats) => {
    setStats(stats);
    if (stats.throughputMbPerSec < 50) {
      setWarning("Input is slow");
    }
  }
});
`,
  }}
/>
